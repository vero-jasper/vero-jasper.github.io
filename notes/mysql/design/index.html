多表#

/*
多表
*/
-- 创建员工表
CREATE TABLE emp(
`id` INT PRIMARY KEY AUTO_INCREMENT,
`name` VARCHAR(20),
age VARCHAR(20),
dept_id INT
);
-- 创建部门表
CREATE TABLE department(
`id` INT PRIMARY KEY AUTO_INCREMENT,
dep_name VARCHAR(30),
dep_location VARCHAR(30)
);
-- 插入部门信息
INSERT INTO department VALUES(NULL, '研发部', '上海'), (NULL, '产品部', '北京');
-- 插入员工信息
INSERT INTO emp (`name`, `age`, dept_id) VALUES ('曹操', 30, 1);
INSERT INTO emp (`name`, `age`, dept_id) VALUES ('荀彧', 28, 1);
INSERT INTO emp (`name`, `age`, dept_id) VALUES ('郭嘉', 26, 1);
INSERT INTO emp (`name`, `age`, dept_id) VALUES ('典韦', 31, 2);
INSERT INTO emp (`name`, `age`, dept_id) VALUES ('许褚', 29, 2);
-- 插入一条不存在部门的数据
INSERT INTO emp (`name`, `age`, dept_id) VALUES ('许褚', 29, 3);
/*
外键约束
作用：外键约束可以让两张表直接产生一个对应的关系，从而保证了主从表引用的完整性
外键：指的是从表中与主表的主键对应的字段
主表和从表
主表 主键id所在的表，一的一方
从表 外键字段所在的表，多的一方
添加外键约束的语法格式
1、创建表的时候添加外键
create table 表名 (
字段...
[constraint] [外键约束名] foreign key(外键字段名) REFERENCES 主表(主表主键字段)
);
*/
CREATE TABLE employee (
eid INT PRIMARY KEY auto_increment,
ename VARCHAR(20),
age INT,
dept_id INT,
CONSTRAINT emp_dept_fk FOREIGN KEY(dept_id) REFERENCES department(id)
);
INSERT INTO employee (`ename`, `age`, dept_id) VALUES ('曹操', 30, 1);
INSERT INTO employee (`ename`, `age`, dept_id) VALUES ('荀彧', 28, 1);
INSERT INTO employee (`ename`, `age`, dept_id) VALUES ('郭嘉', 26, 1);
INSERT INTO employee (`ename`, `age`, dept_id) VALUES ('典韦', 31, 2);
INSERT INTO employee (`ename`, `age`, dept_id) VALUES ('许褚', 29, 2);
-- 插入一条不存在部门的数据，报错
-- 添加外键约束后，会产生一个强制的外键约束检查，保证数据的完整性和一致性
INSERT INTO employee (`ename`, `age`, dept_id) VALUES ('许褚', 29, 3);
/*
删除外键约束
语法格式
ALTER TABLE 从表 drop foreign key 外键约束的名称
*/
ALTER TABLE employee DROP FOREIGN KEY emp_dept_fk;
/*
创建表之后添加外键
ALTER TABLE 从表 ADD foreign key(外键字段名) REFERENCES 主表(主表主键字段)
*/
ALTER TABLE employee add FOREIGN KEY(dept_id) REFERENCES department(id);
/*
外键约束的注意事项
1.从表的外键类型必须与主表的主键类型一致
2.添加数据时，应该先添加主表数据，再去添加从表数据
3.删除数据时，要先删除从表数据，再删主表数据
*/
/*
级联删除
在删除主表数据的同时，可以删除与之相关的从表中的数据
语法
on delete cascade
*/
CREATE TABLE employee (
eid INT PRIMARY KEY auto_increment,
ename VARCHAR(20),
age INT,
dept_id INT,
CONSTRAINT emp_dept_fk FOREIGN KEY(dept_id) REFERENCES department(id),
ON DELETE CASCADE
);
/*
表与表之间的三种关系
一对多的关系（1: n 常见）班级和学生 部门和员工
建表原则 在多的一方建立外键，指向一的一方的主键
多对多的关系（n: n 常见）学生与课程 演员和角色
建表原则 需要创建第三张表(中间表)，中间表中至少要有两个字段，是两张表中的主键字段，作为中间表中的外键
一对一的关系（1: 1 了解）身份证和人
建表原则 可以在任意一方建立外键指向另一方的主键
*/

数据库三范式#

/*
数据库三范式
指数据库的一个设计规则
作用
为了创建冗余较小，结构合理的数据库
范式 是设计数据库的要求(规范)
第一范式(1NF) 满足最低要求的范式
要求 列具有原子性，设计列要做到列不可拆分
第二范式(2NF) 在满足1NF的基础之上，进一步满足更多的规范
要求 一张表只能描述一件事
第三范式(3NF) 在满足2NF的基础之上，以此类推
要求 消除传递依赖，表中的信息如果能够被推导出来，就不要设计一个字段单独记录
三范式就是空间最省原则
*/
/*
反三范式
通过增加冗余或重复数据，来提高数据库性能
浪费存储空间，节省查询时间
冗余字段
某一个字段属于一张表，但他又在多张表中都出现
*/
/*
总结
1.尽量根据三范式规则去设计数据库
2.可以合理的加冗余字段，减少金操作，让数据库执行的更快
*/
-- 例：设计省市表
CREATE TABLE province(
`id` INT PRIMARY KEY auto_increment,
`name` VARCHAR(20),
`descrption` VARCHAR(20)
);
CREATE TABLE city(
`cid` INT PRIMARY KEY auto_increment,
`name` VARCHAR(20),
`descrption` VARCHAR(20),
-- 创建外键，添加外键约束
pid INT,
FOREIGN KEY(pid) REFERENCES province(id)
);
-- 例：设计演员和角色表
/*
多对多
*/
-- 演员表
CREATE TABLE actor (
id INT PRIMARY KEY auto_increment,
`name` VARCHAR(20)
);
-- 角色表
CREATE TABLE role (
id INT PRIMARY KEY auto_increment,
`name` VARCHAR(20)
);
-- 中间表
CREATE TABLE actor_role (
-- 中间表的主键
id INT PRIMARY KEY auto_increment,
-- 外键aid 指向actor表主键
aid INT,
-- rid 指向role表主键
rid INT
);
-- 添加外键约束
-- aid添加外键约束
ALTER TABLE actor_role ADD FOREIGN KEY(aid) REFERENCES actor(id);
ALTER TABLE actor_role ADD FOREIGN KEY(rid) REFERENCES role(id);

多表查询#

/*
多表查询
语法 SELECT 字段列表 FROM 表名列表;
笛卡尔积
所有组合情况
*/
-- 多表查询 交叉连接查询的查询结果会产生笛卡尔积，是不能使用的
SELECT * FROM table1, table2;
/*
1. 内连接查询
特点 通过指定的条件去匹配两张表中的内容，匹配不上的就不显示
隐式内连接
SELECT 字段名 ... FROM 左表, 右表 WHERE 条件
显式内连接
SELECT 字段名 ... FROM 左表 【INNER】 JOIN 右表 on 连接条件
查询确定几件事
1.查询几张表
2.表的连接条件 从表.外键 = 主表.主键
3.查询所用到的字段
4.查询条件
*/
/*
2. 外连接查询
左外连接
语法 关键字 LEFT [OUTER] JOIN
SELECT 字段名 FROM 左表 LEFT JOIN 右表 on 连接条件
特点 以左表为基准，匹配右表中的数据，如果匹配上就显示，否则左表中的数据正常显示，右表数据显示为null
右外连接
语法 关键字 RIGHT [OUTER] JOIN
SELECT 字段名 FROM 左表 RIGHT JOIN 右表 on 连接条件
特点 以右表为基准 匹配左表中的数据 如果匹配上 就显示，否则右表数据正常显示，左表显示null
*/

子查询#

/*
子查询 subQuery
一条select语句的结果，作为另一条select语句的一部分
特点
1. 子查询必须放在小括号中
2. 子查询作为父查询的条件使用
分类
1.WHERE型子查询 将子查询的结果作为父查询的比较条件使用
2.FROM型子查询 将子查询的查询结果作为一张表使用
3.EXISTS型子查询 查询结果是单列多行的情况，可以将子查询的结果作为父查询的in函数中的条件使用
*/
/*
子查询的结果是单列多行，作为父查询的in函数中的条件使用
语法
SELECT 字段名 FROM 表名 WHERE 字段 IN (子查询);
*/
-- eg.
SELECT DISTINCT FROM products WHERE price < 2000;
SELECT * FROM category
WHERE cid IN
(SELECT DISTINCT FROM products WHERE price < 2000);
