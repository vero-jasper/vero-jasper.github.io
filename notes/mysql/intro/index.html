Mysql基础 & SQL入门#

数据库基本概念#

  • 数据库(DataBase) 就是存储和管理数据的仓库
  • 其本质是一个文件系统, 还是以文件的方式,将数据保存在电脑上
存储方式优点缺点
内存速度快不能够永久保存,数据是临时状态的
文件数据是可以永久保存的使用IO流操作文件, 不方便
数据库1.数据可以永久保存 2.方便存储和管理数据 3.使用统一的方式操作数据库(SQL)占用资源,有些数据库需要付费(比如Oracle数据库)

SQL#

概念#

结构化查询语言(Structured Query Language)简称SQL，是一种特殊目的的编程语言，是一种数据库查询和程序设计语言，用于存取数据以及查询、更新和管理关系数据库系统。

作用#

  • 所有关系型数据库有统一查询规范，不同的关系型数据库都支持SQL
  • 所有的关系型数据库都可以使用SQL
  • 不同数据库之间的SQL有一些区别

通用语法#

  • SQL语句可以单行或者多行书写，以分号结尾
  • 可以使用空格和缩进来增加语句的可读性
  • MySql中使用SQL不区分大小写，一般关键字大写，数据库名表名列名小写
  • 注释方式
    • --空格 单行注释
    • /* */ 多行注释
    • '#' Mysql特有的单行注释

SQL的分类#

  • 数据定义语言
    • 简称DDL(Data Definition Language)，用来定义数据库对象：数据库，表，列等。
  • 数据操作语言
    • 简称DML(Data Manipulation Language)，用来对数据库中表的记录进行更新。
  • 数据查询语言
    • 简称DQL(Data Query Language)，用来查询数据库中表的记录。
  • 数据控制语言
    • 简称DCL(Date Control Language)，用来定义数据库的访问权限和安全级别，及创建用户。

DDL操作数据库和数据表#

-- 创建指定名称的数据库
CREATE database db;
CREATE database db character set utf-8;
-- 查看/选择数据库
USE db; -- 切换数据库
SELECT database(); -- 查看当前正在使用的数据库
show databases; --查看Mysql中 都有哪些数据库
show create database 数据库名; -- 查看一个数据库的定义信息
-- 修改数据库
ALTER database db character set utf-8; -- 修改数据库字符集
-- 删除数据库
DROP database db; -- 从MySql中永久的删除某个数据库
-- 创建表
CREATE TABLE test1(
tid INT,
name VARCHAR(20),
PRIMARY KEY (tid)
);
-- 删除表
DROP TABLE IF EXISTS test;
-- 重命名
RENAME TABLE test1 TO test;
-- 修改字符集
ALTER TABLE test CHARACTER SET gbk;
-- 向表中添加字段
ALTER TABLE test ADD cdesc VARCHAR(20);
-- 修改表列的类型或长度 modify
-- 修改字段长度
ALTER TABLE test MODIFY cdesc VARCHAR(50);
-- 修改字段类型
ALTER TABLE test MODIFY cdesc CHAR(20);
-- 修改列的名称
ALTER TABLE test CHANGE c_desc cdesc VARCHAR(30);
-- 删除列
ALTER TABLE test drop cdesc;

DML操作表中数据#

-- DML 对表中数据进行增删改的操作
-- 增加
-- 方式1 插入全部字段，将所有字段名都写出来
INSERT INTO test (tid, `name`, cdesc) VALUES (1, 'vero', 'teacher');
-- 方式2 插入全部字段，不写字段名
INSERT INTO test VALUES(2, '曹操', 'leader');
-- 方式3 插入指定字段
INSERT INTO test (tid, `name`) VALUES (3, '许褚');
-- 注意点
-- 1.值和字段须对应（个数、数据类型、长度）
-- 2.在插入varchar/char/date 类型的时候，必须使用引号包裹
-- 3.如果插入空值 可以忽略不写或写null
-- 修改
-- UPDATE 表名 SET 列名 = 值
-- UPDATE 表名 SET 列名 = 值 [WHERE 条件表达式: 字段名 = 值]
UPDATE test SET name = 'vero jasper' WHERE tid = 1;
-- 一次修改多列
UPDATE test SET name = 'vero', cdesc = '寻玉文若' WHERE tid = 1;
-- 删除
-- 格式1 DELETE FROM 表名;
-- 格式2 DELETE FROM 表名 [WHERE 条件];
DELETE from test WHERE tid = 3;
-- 删除所有数据
DELETE from test; 不推荐，对表中的数据逐条删除
TRUNCATE table 表名; 推荐，删除整张表，然后再创建一个一模一样的新表
TRUNCATE TABLE test;

DQL查询表中数据#

CREATE TABLE emp(
eid INT,
ename VARCHAR(20),
PRIMARY KEY (eid)
);
--
INSERT INTO emp VALUES(3, '曹操');
/*
DQL简单查询
SELECT 列名 FROM 表名;
*/
SELECT * FROM emp; -- *表示所有列
SELECT eid, ename FROM emp;
-- 查询所有数据，然后将列名改为中文
-- 别名查询
-- AS可以省略
SELECT
eid AS '编号',
ename '姓名'
FROM emp;
-- 去重操作 关键字 DISTINCT
SELECT DISTINCT ename FROM emp;
-- 不会对数据进行修改
SELECT ename, eid+1 FROM emp;
/*
条件查询
SELECT 列名 FROM 表名 WHERE 条件表达式
比较运算符
> < >= <= = <> !=
BETWEEN ... AND ...
IN(集合)
LIKE
IS NULL
逻辑运算符
AND
OR
NOT
*/
SELECT * FROM emp WHERE ename = 'vero';
/*
LIKE '%e%'
% 通配符，表示匹配任意多个字符串
_ 表示匹配一个字符
*/
SELECT * FROM emp WHERE ename LIKE '%e%';

DQL排序查询#

/*
排序
使用 ORDER BY
SELECT 字段名 FROM 表名 [WHERE 字段名 = 值] ORDER BY 字段名称 [ASC/DESC];
ASC 升序排序（默认）
DESC 降序排序
*/
-- 单列排序 按照某一个字段进行排序
SELECT * FROM emp ORDER BY ename DESC;
-- 组合排序
SELECT * FROM emp ORDER BY ename DESC, eid;
/*
聚合函数
作用：将一列数据作为一个整体，进行纵向的计算
常用的聚合函数
count(字段) 统计记录数
sum(字段) 求和
max(字段) 求最大值
min(字段) 求最小值
avg(字段) 求平均值
*/
-- count 在统计时会忽略空值
SELECT COUNT(*) FROM emp;
SELECT
SUM(eid) AS '编号',
MAX(eid) '最大值',
MIN(eid) '最小值',
AVG(eid) '平均值'
FROM emp;
/*
分组查询
GROUP BY
SELECT 分组字段/聚合函数 FROM 表名 GROUP BY 分组字段
*/
SELECT ename, AVG(eid) FROM emp GROUP BY ename;
SELECT ename AS '姓名' FROM emp GROUP BY ename;
-- 在分组之后，进行条件过滤， 使用: having 判断条件
/*
WHERE与Having区别
WHERE
1. 在分组前进行过滤
2. WHERE 后面不能跟 聚合函数
HAVING
1. 在分组之后进行条件过滤
2. HAVING后面可以写聚合函数
*/
/*
LIMIT 通过limit 去指定要查询的数据的条数和行数
语法格式
SELECT 字段 FROM 表名 LIMIT offset, length;
参数说明
offset: 起始行数 默认从0开始
length: 返回行数（查询几条数据）
*/
SELECT * FROM emp LIMIT 1, 2;
/*
约束
指对数据进行一定的限制，来保证数据的完整性、有效性、正确性
常见的约束
主键约束 PRIMARY KEY
唯一约束 UNIQUE
非空约束 NOT NULL
外键约束 FOREIGN KEY
*/
/*
主键约束
特点: 不可重复 唯一 非空
作用: 用来表示数据库中的每一条记录
语法格式
字段名 字段类型 PRIMARY KEY;
*/
-- 1. 创建带有主键的表
CREATE TABLE emp2(
eid INT PRIMARY KEY,
ename VARCHAR(20),
sex CHAR(1)
);
-- DESC emp2;
-- 2.
CREATE TABLE emp2(
eid INT,
ename VARCHAR(20),
sex CHAR(1),
PRIMARY KEY (eid)
);
-- DESC emp2;
-- 3.创建表之后再添加主键
CREATE TABLE emp2(
eid INT,
ename VARCHAR(20),
sex CHAR(1)
);
ALTER TABLE emp2 add PRIMARY KEY (eid);
-- DESC emp2;
-- 删除主键
ALTER TABLE emp2 DROP PRIMARY KEY;
-- DESC emp2;
/*
主键的自增
关键字: auto_increment 主键自动增长，字段必须是整数类型
*/
-- 创建主键自增的表
CREATE TABLE emp2(
eid INT PRIMARY KEY auto_increment,
ename VARCHAR(20),
sex char(1)
);
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2 VALUES(null, '曹操', '男');
-- 修改自增起始值
CREATE TABLE emp2(
eid INT PRIMARY KEY auto_increment,
ename VARCHAR(20),
sex char(1)
) auto_increment=100;
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2 VALUES(null, '曹操', '男');
/*
DELETE 和 TRUNCATE 对自增的影响
DELETE 将表中所有数据逐条删除
TRUNCATE 将整个表删除再创建一个结构相同的表
*/
-- delete 方式删除所有数据，对自增没有影响
DELETE FROM emp2;
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2 VALUES(null, '曹操', '男');
truncate 删除所有数据，主键自增从1开始
TRUNCATE TABLE emp2;
INSERT INTO emp2(ename, sex) VALUES('曹操', '男');
INSERT INTO emp2 VALUES(null, '曹操', '男');
/*
非空约束
特点 某一列不允许为空
语法格式
字段名 字段类型 NOT NULL;
*/
CREATE TABLE emp2(
int id PRIMARY KEY AUTO_INCREMENT,
ename VARCHAR(20) NOT NULL
);
/*
唯一约束
特点: 表中某一列不能重复(NULL 不做唯一判断)
语法格式
字段名 字段类型 unique
*/
CREATE TABLE emp2(
int id PRIMARY KEY AUTO_INCREMENT,
ename VARCHAR(20) UNIQUE
);
/*
主键约束和唯一约束的区别
1. 主键约束 唯一不能为空
2. 唯一约束 唯一但可以为空
3. 一个表中只能有一个主键，但是可以有多个唯一约束
外键约束（多表中学习）
*/
